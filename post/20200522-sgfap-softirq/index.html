гҖҠйқўеҗ‘еә”з”ЁејҖеҸ‘иҖ…зҡ„зі»з»ҹжҢҮеҚ—гҖӢCPUзҜҮд№ӢиҪҜдёӯж–ӯ

2020е№ҙ5жңҲ22ж—Ҙ

жң¬ж–ҮжҳҜгҖҠйқўеҗ‘еә”з”ЁејҖеҸ‘иҖ…зҡ„зі»з»ҹжҢҮеҚ—гҖӢж–ҮжЎЈе…¶дёӯзҡ„дёҖзҜҮпјҢе®Ңж•ҙзҡ„зӣ®еҪ•и§ҒгҖҠйқўеҗ‘еә”з”ЁејҖеҸ‘иҖ…зҡ„зі»з»ҹжҢҮеҚ—гҖӢеҜји®әгҖӮ

жҰӮиҝ° #

дёӯж–ӯпјҲinterruptпјүйҖҡеёёиў«е®ҡд№үдёәдёҖдёӘдәӢд»¶пјҢиҜҘдәӢд»¶ж”№еҸҳеӨ„зҗҶеҷЁжү§иЎҢзҡ„жҢҮд»ӨйЎәеәҸгҖӮдёӯж–ӯеҲҶдёәеҗҢжӯҘе’ҢејӮжӯҘдёӨз§Қпјҡ

  • еҗҢжӯҘдёӯж–ӯеңЁжҢҮд»Өжү§иЎҢж—¶з”ұCPUжҺ§еҲ¶еҚ•е…ғдә§з”ҹпјҢд№ӢжүҖд»Ҙз§°дёәеҗҢжӯҘпјҢжҳҜеӣ дёәеҸӘжңүеңЁдёҖжқЎжҢҮд»Өз»Ҳжӯўжү§иЎҢеҗҺCPUжүҚеҸ‘з”ҹдёӯж–ӯгҖӮ
  • ејӮжӯҘдёӯж–ӯжҳҜз”ұе…¶д»–зЎ¬д»¶и®ҫеӨҮдҫқз…§CPUж—¶й’ҹдҝЎеҸ·йҡҸжңәдә§з”ҹзҡ„гҖӮ

еңЁIntelзҡ„еӨ„зҗҶеҷЁжүӢеҶҢдёӯпјҢе°ҶеҗҢжӯҘдёӯж–ӯз§°дёәвҖңејӮеёёпјҲexceptionпјүвҖқпјҢејӮжӯҘдёӯж–ӯз§°дёәвҖңдёӯж–ӯвҖқгҖӮ

ејӮеёёйҖҡеёёз”ұзЁӢеәҸзҡ„й”ҷиҜҜдә§з”ҹпјҢжҲ–иҖ…жҳҜз”ұеҶ…ж ёеҝ…йЎ»еӨ„зҗҶзҡ„ејӮеёёжқЎд»¶дә§з”ҹзҡ„гҖӮжҜ”еҰӮзЁӢеәҸдёӯжңүйҷӨйӣ¶ејӮеёёпјҢжҜ”еҰӮиҝӣзЁӢиҝҗиЎҢиҝҮзЁӢдёӯдә§з”ҹзҡ„вҖңзјәйЎөејӮеёёпјҲpagefaultпјүвҖқзӯүпјҢйғҪеұһдәҺејӮеёёгҖӮ

иҖҢдёӯж–ӯжҳҜз”ұе®ҡж—¶еҷЁе’ҢI/Oи®ҫеӨҮдә§з”ҹзҡ„пјҢжҜ”еҰӮз”ЁжҲ·зҡ„дёҖж¬ЎжҢүй”®гҖҒзҪ‘еҚЎж”¶еҲ°ж•°жҚ®пјҢйғҪдјҡдә§з”ҹдёӯж–ӯгҖӮ

еӨ„зҗҶеҷЁдёҖж—Ұж”¶еҲ°дёӯж–ӯпјҢе°ұеҝ…йЎ»жү“ж–ӯеҪ“еүҚзҡ„жү§иЎҢпјҢиҪ¬иҖҢеҺ»жү§иЎҢдёӯж–ӯеӨ„зҗҶеҮҪж•°гҖӮдёӯж–ӯеӨ„зҗҶеҮҪж•°пјҢжң¬иә«жңүдёҖдәӣзјәйҷ·пјҡ

  • дёҚиғҪеңЁиҝӣзЁӢдёҠдёӢж–Үдёӯжү§иЎҢпјҢеӣ жӯӨдёҚиғҪйҳ»еЎһгҖӮ
  • дёӯж–ӯеӨ„зҗҶзЁӢеәҸдјҡжү“ж–ӯзЁӢеәҸжү§иЎҢпјҢдёәдәҶйҒҝе…ҚиҝҷдёӘжү“ж–ӯзҡ„жөҒзЁӢеҒңжӯўж—¶й—ҙиҝҮй•ҝпјҢжүҖд»Ҙеә”иҜҘжү§иЎҢзҡ„и¶Ҡзҹӯи¶ҠеҘҪгҖӮ

еӣ дёәд»ҘдёҠзҡ„еҺҹеӣ пјҢLinuxеҶ…ж ёе°Ҷдёӯж–ӯзҡ„еӨ„зҗҶеҲҶдёәдәҶдёҠдёӢдёӨйғЁеҲҶпјҢе…¶дёӯдёҠеҚҠйғЁе°ұжҳҜеүҚйқўжҸҗеҲ°зҡ„дёӯж–ӯеӨ„зҗҶеҮҪж•°пјҢиҝҷйғЁеҲҶиғҪеӨҹжңҖеҝ«зҡ„е“Қеә”дёӯж–ӯпјҢе№¶дё”еҒҡдёҖдәӣдёӯж–ӯеҗҺеҝ…йЎ»иҰҒеҒҡзҡ„дәӢжғ…пјҢиҖҢдёҖдәӣеҸҜд»ҘеңЁдёӯж–ӯеӨ„зҗҶеҮҪж•°еҗҺз»§з»ӯжү§иЎҢзҡ„ж“ҚдҪңпјҢеҲҷеҸҜд»Ҙж”ҫеңЁдёӢеҚҠйғЁдёӯгҖӮ

д»ҘзҪ‘еҚЎжҺҘж”¶еҲ°ж•°жҚ®жқҘдёҫдҫӢпјҢзҪ‘еҚЎйҖҡиҝҮдёӯж–ӯе‘ҠиҜүеҶ…ж ёжңүж•°жҚ®еҸҜд»ҘжҺҘж”¶пјҢжӯӨж—¶еҶ…ж ёе°ұдјҡеҲ°зҪ‘еҚЎзҡ„дёӯж–ӯеӨ„зҗҶзЁӢеәҸдёӯжү§иЎҢдёҖдәӣзҪ‘еҚЎзЎ¬д»¶зҡ„еҝ…иҰҒи®ҫзҪ®пјҢиҖҢеҜ№еә”зҡ„дёӢеҚҠйғЁе°ұжҳҜеӨ„зҗҶзҪ‘еҚЎж”¶еҲ°зҡ„ж•°жҚ®пјҢеӣ дёәеӨ„зҗҶзҪ‘еҚЎж•°жҚ®жІЎжңүеҝ…иҰҒеңЁдёӯж–ӯеӨ„зҗҶеҮҪж•°йҮҢйқўй©¬дёҠжү§иЎҢгҖӮ

дёӨиҖ…зҡ„дё»иҰҒеҢәеҲ«еңЁдәҺпјҡдёӯж–ӯдёҚиғҪиў«зӣёеҗҢзұ»еһӢзҡ„дёӯж–ӯжү“ж–ӯпјҢиҖҢдёӢеҚҠйғЁдҫқз„¶еҸҜд»Ҙиў«дёӯж–ӯжү“ж–ӯпјӣдёӯж–ӯеҜ№дәҺж—¶й—ҙйқһеёёж•Ҹж„ҹпјҢиҖҢдёӢеҚҠйғЁеҹәжң¬дёҠйғҪжҳҜдёҖдәӣеҸҜд»Ҙе»¶иҝҹзҡ„е·ҘдҪңгҖӮз”ұдәҺдәҢиҖ…зҡ„иҝҷз§ҚеҢәеҲ«пјҢжүҖд»ҘеҜ№дәҺдёҖдёӘе·ҘдҪңжҳҜж”ҫеңЁдёҠеҚҠйғЁиҝҳжҳҜж”ҫеңЁдёӢеҚҠйғЁеҺ»жү§иЎҢпјҢеҸҜд»ҘеҸӮиҖғдёӢйқў4жқЎпјҡ

  • еҰӮжһңдёҖдёӘд»»еҠЎеҜ№ж—¶й—ҙйқһеёёж•Ҹж„ҹпјҢе°Ҷе…¶ж”ҫеңЁдёӯж–ӯеӨ„зҗҶзЁӢеәҸдёӯжү§иЎҢгҖӮ
  • еҰӮжһңдёҖдёӘд»»еҠЎе’ҢзЎ¬д»¶зӣёе…іпјҢе°Ҷе…¶ж”ҫеңЁдёӯж–ӯеӨ„зҗҶзЁӢеәҸдёӯжү§иЎҢгҖӮ
  • еҰӮжһңдёҖдёӘд»»еҠЎиҰҒдҝқиҜҒдёҚиў«е…¶д»–дёӯж–ӯпјҲзү№еҲ«жҳҜзӣёеҗҢзҡ„дёӯж–ӯпјүжү“ж–ӯпјҢе°Ҷе…¶ж”ҫеңЁдёӯж–ӯеӨ„зҗҶзЁӢеәҸдёӯжү§иЎҢгҖӮ
  • е…¶д»–жүҖжңүд»»еҠЎпјҢиҖғиҷ‘ж”ҫеңЁдёӢеҚҠйғЁеҺ»жү§иЎҢгҖӮ

жңүеҶҷеҶ…ж ёд»»еҠЎйңҖиҰҒе»¶еҗҺжү§иЎҢпјҢеӣ жӯӨжүҚжңүзҡ„дёӢеҚҠйғЁпјҢиҝӣиҖҢе®һзҺ°дәҶдёүз§Қе®һзҺ°дёӢеҚҠйғЁзҡ„ж–№жі•гҖӮиҝҷе°ұжҳҜжң¬ж–ҮиҰҒи®Ёи®әзҡ„иҪҜдёӯж–ӯгҖҒtaskletе’Ңе·ҘдҪңйҳҹеҲ—гҖӮ

иҪҜдёӯж–ӯ #

иҪҜдёӯж–ӯдҪңдёәдёӢеҚҠйғЁжңәеҲ¶зҡ„д»ЈиЎЁпјҢжҳҜйҡҸзқҖSMPпјҲshare memory processorпјүзҡ„еҮәзҺ°еә”иҝҗиҖҢз”ҹзҡ„пјҢе®ғд№ҹжҳҜtaskletе®һзҺ°зҡ„еҹәзЎҖпјҲtaskletе®һйҷ…дёҠеҸӘжҳҜеңЁиҪҜдёӯж–ӯзҡ„еҹәзЎҖдёҠж·»еҠ дәҶдёҖе®ҡзҡ„жңәеҲ¶пјүгҖӮиҪҜдёӯж–ӯдёҖиҲ¬жҳҜвҖңеҸҜе»¶иҝҹеҮҪж•°вҖқзҡ„жҖ»з§°пјҢжңүж—¶еҖҷд№ҹеҢ…жӢ¬дәҶtaskletпјҲиҜ·иҜ»иҖ…еңЁйҒҮеҲ°зҡ„ж—¶еҖҷж №жҚ®дёҠдёӢж–ҮжҺЁж–ӯжҳҜеҗҰеҢ…еҗ«taskletпјүгҖӮе®ғзҡ„еҮәзҺ°е°ұжҳҜеӣ дёәиҰҒж»Ўи¶ідёҠйқўжүҖжҸҗеҮәзҡ„дёҠеҚҠйғЁе’ҢдёӢеҚҠйғЁзҡ„еҢәеҲ«пјҢдҪҝеҫ—еҜ№ж—¶й—ҙдёҚж•Ҹж„ҹзҡ„д»»еҠЎе»¶еҗҺжү§иЎҢпјҢиҖҢдё”еҸҜд»ҘеңЁеӨҡдёӘCPUдёҠе№¶иЎҢжү§иЎҢпјҢдҪҝеҫ—жҖ»зҡ„зі»з»ҹж•ҲзҺҮеҸҜд»Ҙжӣҙй«ҳгҖӮе®ғзҡ„зү№жҖ§еҢ…жӢ¬пјҡ

дә§з”ҹеҗҺе№¶дёҚжҳҜй©¬дёҠеҸҜд»Ҙжү§иЎҢпјҢеҝ…йЎ»иҰҒзӯүеҫ…еҶ…ж ёзҡ„и°ғеәҰжүҚиғҪжү§иЎҢгҖӮиҪҜдёӯж–ӯдёҚиғҪиў«иҮӘе·ұжү“ж–ӯ(еҚіеҚ•дёӘcpuдёҠиҪҜдёӯж–ӯдёҚиғҪеөҢеҘ—жү§иЎҢ)пјҢеҸӘиғҪиў«зЎ¬д»¶дёӯж–ӯжү“ж–ӯпјҲдёҠеҚҠйғЁпјүгҖӮ еҸҜд»Ҙе№¶еҸ‘иҝҗиЎҢеңЁеӨҡдёӘCPUдёҠпјҲеҚідҪҝеҗҢдёҖзұ»еһӢзҡ„д№ҹеҸҜд»ҘпјүгҖӮжүҖд»ҘиҪҜдёӯж–ӯеҝ…йЎ»и®ҫи®ЎдёәеҸҜйҮҚе…Ҙзҡ„еҮҪж•°пјҲе…Ғи®ёеӨҡдёӘCPUеҗҢж—¶ж“ҚдҪңпјүпјҢеӣ жӯӨд№ҹйңҖиҰҒдҪҝз”ЁиҮӘж—Ӣй”ҒжқҘдҝқе…¶ж•°жҚ®з»“жһ„гҖӮ

ж•°жҚ®з»“жһ„ #

иҪҜдёӯж–ӯз”ұзі»з»ҹеңЁеҗҜеҠЁзҡ„ж—¶еҖҷжіЁеҶҢеҲ°еҶ…ж ёдёӯпјҢз”ұдёҖдёӘе…ЁеұҖж•°з»„жқҘз»ҙжҠӨиҪҜдёӯж–ӯпјҡ

struct softirq_action
{
	void	(*action)(struct softirq_action *);
};

static struct softirq_action softirq_vec[NR_SOFTIRQS] __cacheline_aligned_in_smp;

еҸҜд»ҘзңӢеҲ°пјҢжң¬иҙЁдёҠз»“жһ„дҪ“softirq_actionеӯҳеӮЁзҡ„жҳҜеҮҪж•°жҢҮй’ҲиҖҢе·ІпјҢиҪҜдёӯж–ӯжңүд»ҘдёӢзұ»еһӢпјҡ

enum
{
	HI_SOFTIRQ=0,     // еӨ„зҗҶй«ҳдјҳе…Ҳзә§зҡ„tasklet
	TIMER_SOFTIRQ,    // е®ҡж—¶еҷЁзҡ„дёӢеҚҠйғЁ
	NET_TX_SOFTIRQ,   // зҪ‘еҚЎеҸ‘йҖҒж•°жҚ®еҢ…
	NET_RX_SOFTIRQ,   // зҪ‘еҚЎжҺҘж”¶ж•°жҚ®еҢ…
	BLOCK_SOFTIRQ,    // BLOCKиЈ…зҪ®     
	IRQ_POLL_SOFTIRQ, 
	TASKLET_SOFTIRQ,  // еӨ„зҗҶеёёи§„зҡ„tasklet
	SCHED_SOFTIRQ,
	HRTIMER_SOFTIRQ, 
	RCU_SOFTIRQ,
	NR_SOFTIRQS
};

зі»з»ҹжҸҗдҫӣдәҶopen_softirqеҮҪж•°з”ЁдәҺеҗ„дёӘйңҖиҰҒдҪҝз”ЁеҲ°иҪҜдёӯж–ӯзҡ„зі»з»ҹжіЁеҶҢеҜ№еә”зҡ„иҪҜдёӯж–ӯеӨ„зҗҶеҮҪж•°гҖӮ

void open_softirq(int nr, void (*action)(struct softirq_action *))
{
	softirq_vec[nr].action = action;
}

еҗҢж—¶пјҢиҝҳжҸҗдҫӣдәҶsoftirq_to_nameж•°з»„пјҢз”ЁдәҺжҠҠиҪҜдёӯж–ӯзҡ„зҙўеј•жҳ е°„еҲ°еҜ№еә”зҡ„иҪҜдёӯж–ӯеҗҚз§°пјҡ

const char * const softirq_to_name[NR_SOFTIRQS] = {
	"HI", "TIMER", "NET_TX", "NET_RX", "BLOCK", "IRQ_POLL",
	"TASKLET", "SCHED", "HRTIMER", "RCU"
};

еңЁLinuxдёӢйқўпјҢеҸҜд»ҘйҖҡиҝҮжҹҘзңӢ/proc/softirqsж–Үд»¶зҹҘйҒ“еҪ“еүҚзі»з»ҹиҪҜдёӯж–ӯзҡ„жғ…еҶөпјҡ

и§ҰеҸ‘иҪҜдёӯж–ӯ #

и§ҰеҸ‘иҪҜдёӯж–ӯзҡ„е…ҘеҸЈеҮҪж•°жҳҜraise_softirqпјҡ

void raise_softirq(unsigned int nr)
{
	unsigned long flags;

	local_irq_save(flags);
	raise_softirq_irqoff(nr);
	local_irq_restore(flags);
}

йҰ–е…ҲпјҢз”ұдәҺи°ғз”ЁиҪҜдёӯж–ӯеӨ„зҗҶеҮҪж•°еҝ…йЎ»зҰҒз”Ёдёӯж–ӯпјҢжүҖд»Ҙlocal_irq_saveе®Ҹе°ҶдҝқеӯҳеңЁeflagsеҜ„еӯҳеҷЁдёӯзҡ„IFж Үеҝ—пјҢеҗҢж—¶зҰҒз”Ёжң¬ең°еӨ„зҗҶеҷЁзҡ„дёӯж–ӯгҖӮеҜ№еә”зҡ„пјҢlocal_irq_restoreе®Ҹе°ҶдҝқеӯҳдёӢжқҘзҡ„flagsж Үеҝ—дҪҚжҒўеӨҚеӣһеҺ»пјҢеҗҢж—¶жү“ејҖжң¬ең°еӨ„зҗҶеҷЁзҡ„дёӯж–ӯгҖӮ

жҺҘзқҖзңӢraise_softirq_irqoffеҮҪж•°пјҡ

inline void raise_softirq_irqoff(unsigned int nr)
{
	__raise_softirq_irqoff(nr);

	if (!in_interrupt())
		wakeup_softirqd();
}

  • йҰ–е…ҲпјҢи°ғз”Ё__raise_softirq_irqoffе°Ҷжң¬ең°еӨ„зҗҶеҷЁзҡ„дёӯ__softirq_pendingеҸҳйҮҸеҜ№еә”nrиҝҷдёӘиҪҜдёӯж–ӯзҡ„дҪҚзҪ®дёә1пјҢиЎЁзӨәиҜҘзұ»еһӢзҡ„иҪҜдёӯж–ӯиў«и§ҰеҸ‘дәҶгҖӮ

  • з„¶еҗҺпјҢйҖҡиҝҮin_interruptеҮҪж•°жқҘеҲӨж–ӯеӨ„зҗҶеҷЁеҪ“еүҚжҳҜеҗҰеңЁеӨ„зҗҶдёӯж–ӯзҠ¶жҖҒпјҢеҰӮжһңжІЎжңүеҲҷи°ғз”Ёwakeup_softirqdеҮҪж•°е”ӨйҶ’жң¬еӨ„зҗҶеҷЁзҡ„ksoftirqdеҶ…ж ёиҝӣзЁӢпјҡ

static void wakeup_softirqd(void)
{
	struct task_struct *tsk = __this_cpu_read(ksoftirqd);

	if (tsk && tsk->state != TASK_RUNNING)
		wake_up_process(tsk);
}

еңЁLinuxзі»з»ҹдёӯпјҢжҜҸдёӘCPUйғҪдјҡиҝҗиЎҢдёҖдёӘksoftirqdеҶ…ж ёиҝӣзЁӢпјҢдё“й—Ёз”ұиҝҷдёӘиҝӣзЁӢжқҘеӨ„зҗҶжң¬CPUзҡ„иҪҜдёӯж–ӯгҖӮksoftirqdеҶ…ж ёиҝӣзЁӢзҡ„дё»иҰҒжөҒзЁӢеңЁеҮҪж•°__do_softirqдёӯе®ҢжҲҗпјҢе…¶ж ёеҝғжҳҜдёҖдёӘеҫӘзҺҜпјҢз”ЁжқҘжЈҖжөӢеҪ“еүҚжңүе“ӘдәӣжңӘеӨ„зҗҶзҡ„иҪҜдёӯж–ӯпјҢи°ғз”Ёе…¶жіЁеҶҢзҡ„еӨ„зҗҶеҮҪж•°жқҘеӨ„зҗҶиҪҜдёӯж–ӯпјҡ

asmlinkage __visible void __softirq_entry __do_softirq(void)
{

		trace_softirq_entry(vec_nr);	// и®°еҪ•ејҖе§ӢеӨ„зҗҶиҪҜдёӯж–ӯзҡ„trace event
		h->action(h);									// и°ғз”ЁжіЁеҶҢзҡ„иҪҜдёӯж–ӯеӨ„зҗҶеҮҪж•°
		trace_softirq_exit(vec_nr);		// и®°еҪ•з»“жқҹеӨ„зҗҶиҪҜдёӯж–ӯзҡ„trace event
	}

жҖ»з»“жқҘиҜҙпјҢиҰҒи§ҰеҸ‘дёҖдёӘиҪҜдёӯж–ӯеҸӘйңҖиҰҒд»ҘдёӢеҮ жӯҘпјҡ

  1. дҝқеӯҳIFж Үеҝ—пјҢзҰҒз”Ёдёӯж–ӯгҖӮ
  2. е°ҶиҝҷдёӘиҪҜдёӯж–ӯеҜ№еә”зҡ„дҪҚзҪ®дёә1гҖӮ

  • еҸӘиғҪеңЁзі»з»ҹеҗҜеҠЁж—¶жіЁеҶҢпјҢеҸҰеӨ–ж•°йҮҸе’Ңзұ»еһӢдёҚиғҪеҠЁжҖҒеҸҳжӣҙгҖӮ
  • еӣ дёәжҜҸдёӘеӨ„зҗҶеҷЁдёҠйғҪжңүдёҖдёӘksoftirqdеҶ…ж ёиҝӣзЁӢпјҢеҸҜиғҪеҗҢж—¶еңЁеӨ„зҗҶеҗҢдёҖз§Қзұ»еһӢзҡ„иҪҜдёӯж–ӯпјҢиҪҜдёӯж–ӯеҝ…йЎ»е®һзҺ°дёәеҸҜйҮҚе…ҘеҮҪж•°пјҢеҜјиҮҙејҖеҸ‘дёҠзҡ„еӨҚжқӮеәҰжҸҗй«ҳгҖӮеҰӮжһңжҹҗз§Қеә”з”Ёе№¶дёҚйңҖиҰҒеңЁеӨҡдёӘCPUдёҠе№¶иЎҢжү§иЎҢпјҢйӮЈд№ҲиҪҜдёӯж–ӯжҳҜжІЎжңүеҝ…иҰҒзҡ„гҖӮ

  • дёҖз§Қзұ»еһӢзҡ„taskletеҸӘиғҪиҝҗиЎҢеңЁдёҖдёӘCPUдёҠпјҢдёҚиғҪе№¶иЎҢиҖҢеҸӘиғҪдёІиЎҢжү§иЎҢгҖӮ
  • еӨҡдёӘдёҚеҗҢзұ»еһӢзҡ„taskletеҸҜд»Ҙе№¶иЎҢеңЁеӨҡдёӘCPUдёҠгҖӮ
  • иҪҜдёӯж–ӯжҳҜйқҷжҖҒпјҢеҸӘиғҪж”ҜжҢҒжңүйҷҗзҡ„еҮ з§ҚиҪҜдёӯж–ӯзұ»еһӢпјҢдёҖж—ҰеҶ…ж ёзј–иҜ‘еҘҪд№ӢеҗҺе°ұдёҚиғҪж”№еҸҳпјӣиҖҢtaskletзҒөжҙ»еҫҲеӨҡпјҢеҸҜд»ҘйҖҡиҝҮж·»еҠ еҶ…ж ёжЁЎеқ—зҡ„ж–№ејҸеңЁиҝҗиЎҢж—¶дҝ®ж”№гҖӮ

taskletжҳҜеңЁдёӨз§ҚиҪҜдёӯж–ӯзұ»еһӢзҡ„еҹәзЎҖдёҠе®һзҺ°зҡ„пјҢеӣ жӯӨеҰӮжһңдёҚйңҖиҰҒиҪҜдёӯж–ӯзҡ„е№¶иЎҢзү№жҖ§пјҢtaskletе°ұжҳҜжңҖеҘҪзҡ„йҖүжӢ©гҖӮд№ҹе°ұжҳҜиҜҙtaskletжҳҜиҪҜдёӯж–ӯзҡ„дёҖз§Қзү№ж®Ҡз”Ёжі•пјҢеҚіе»¶иҝҹжғ…еҶөдёӢзҡ„дёІиЎҢжү§иЎҢгҖӮ

йҰ–е…ҲжқҘзңӢеҲқе§ӢеҢ–пјҢеҜ№еә”зҡ„жҳҜsoftirq_initеҮҪж•°пјҡ

void __init softirq_init(void)
{
	int cpu;

	for_each_possible_cpu(cpu) {
		per_cpu(tasklet_vec, cpu).tail =
			&per_cpu(tasklet_vec, cpu).head;
		per_cpu(tasklet_hi_vec, cpu).tail =
			&per_cpu(tasklet_hi_vec, cpu).head;
	}

	open_softirq(TASKLET_SOFTIRQ, tasklet_action);
	open_softirq(HI_SOFTIRQ, tasklet_hi_action);
}

еҸҜд»ҘзңӢеҲ°пјҢtaskletеҗ„жңүдёҖдёӘеҹәдәҺTASKLET_SOFTIRQе’ҢHI_SOFTIRQдёӨдёӘзұ»еһӢиҪҜдёӯж–ӯзҡ„taskletпјҢиҝҷдёӨдёӘзұ»еһӢеҲҶеҲ«еҜ№еә”дҪҺдјҳе…Ҳзә§е’Ңй«ҳдјҳе…Ҳзә§зҡ„taskletпјҢеҸҳйҮҸtasklet_vecе’Ңtasklet_hi_vecзҡ„е®ҡд№үеҰӮдёӢпјҡ

struct tasklet_head {
	struct tasklet_struct *head;
	struct tasklet_struct **tail;

static DEFINE_PER_CPU(struct tasklet_head, tasklet_vec);

иҝҷдёӨдёӘеҸҳйҮҸеҲҶеҲ«е®ҡд№үдәҶдёӨдёӘеӯҳеӮЁtasklet_structз»“жһ„дҪ“зҡ„й“ҫиЎЁгҖӮз»“жһ„дҪ“tasklet_structе®ҡд№үеҰӮдёӢпјҡ

{

е…¶дёӯеҢ…еҗ«дәҶдә”дёӘжҲҗе‘ҳеҸҳйҮҸпјҢеҲҶеҲ«жҳҜпјҡ

  • struct tasklet_struct *nextпјҡй“ҫиЎЁдёӯдёӢдёҖдёӘжҲҗе‘ҳзҡ„жҢҮй’ҲгҖӮ

йҰ–е…ҲпјҢдҪҝз”Ёtest_and_set_bitеҮҪж•°еҲӨж–ӯt->stateдёӯзҡ„TASKLET_STATE_SCHEDжҳҜеҗҰиў«зҪ®дёә1пјҢеҰӮжһңжІЎжңүеҲҷи°ғз”ЁеҮҪж•°__tasklet_scheduleи°ғеәҰtaskletжқҘжү§иЎҢпјҡ

// kernel/softirq.c
void __tasklet_schedule(struct tasklet_struct *t)
{
	unsigned long flags;

	local_irq_save(flags);
	t->next = NULL;
	*__this_cpu_read(tasklet_vec.tail) = t;
	__this_cpu_write(tasklet_vec.tail, &(t->next));
	raise_softirq_irqoff(TASKLET_SOFTIRQ);
	local_irq_restore(flags);
}

иҜҘеҮҪж•°йҰ–е…Ҳе’ҢеүҚйқўиҪҜдёӯж–ӯзҡ„еӨ„зҗҶдёҖж ·пјҢдҝқеӯҳIFж Үеҝ—дҪҚеҗҢж—¶зҰҒз”Ёдёӯж–ӯпјҢз„¶еҗҺдҝ®ж”№tasklet_vecй“ҫиЎЁе°Ҷж–°еўһзҡ„taskletж·»еҠ еҲ°й“ҫиЎЁе°ҫйғЁпјҢи°ғз”Ёraise_softirq_irqoffи§ҰеҸ‘TASKLET_SOFTIRQзұ»еһӢзҡ„иҪҜдёӯж–ӯжү§иЎҢпјҢжңҖеҗҺжҒўеӨҚIFж Үеҝ—дҪҚеҗҢж—¶еҗҜз”Ёдёӯж–ӯгҖӮ

еңЁеҮҪж•°softirq_initдёӯпјҢдёӨзұ»з”ЁдәҺеӨ„зҗҶtaskletзҡ„еӨ„зҗҶеҮҪж•°еҲҶеҲ«жҳҜпјҡ

// kernel/softirq.c
static __latent_entropy void tasklet_action(struct softirq_action *a)
{
	struct tasklet_struct *list;

	while (list) {
		struct tasklet_struct *t = list;
		t->func(t->data);
	}
}

// kernel/workqueue.c
bool queue_work_on(int cpu, struct workqueue_struct *wq,
		   struct work_struct *work)
{
	bool ret = false;
	unsigned long flags;

	local_irq_save(flags);

	if (!test_and_set_bit(WORK_STRUCT_PENDING_BIT, work_data_bits(work))) {
		__queue_work(cpu, wq, work);
		ret = true;
	}

	local_irq_restore(flags);
	return ret;
}

{

	if (req_cpu == WORK_CPU_UNBOUND)
		cpu = wq_select_unbound_cpu(raw_smp_processor_id());

	if (!(wq->flags & WQ_UNBOUND))
		pwq = per_cpu_ptr(wq->cpu_pwqs, cpu);
	else
		pwq = unbound_pwq_by_node(wq, cpu_to_node(cpu));

	// ...

	insert_work(pwq, work, worklist, work_flags);

	// ...
}

иҝҷйҮҢиЈҒеүӘжҺүж— е…ізҡ„д»Јз ҒпјҢжқҘзңӢж ёеҝғзҡ„йғЁеҲҶпјҡ

  • йҰ–е…ҲпјҢеҰӮжһңworkеҠ е…Ҙж—¶жңӘжҢҮе®ҡиҰҒиҝҗиЎҢзҡ„CPUпјҢйҖҡиҝҮwq_select_unbound_cpuиҝӣиЎҢйҖүжӢ©пјҢй»ҳи®ӨдҪҝз”ЁеҪ“еүҚCPUгҖӮеҰӮжһңиҜҘCPUдёҚеңЁwq_unbound_cpumask (е…ЁеұҖ cpumask)еҶ…пјҢеҲҷд»Һwq_unbound_cpumaskдёӯйҖҡиҝҮround robinж–№ејҸйҖүжӢ©гҖӮ
  • еҜ№дәҺ bound workqueue пјҢеҸ–еҮәеҪ“еүҚ per CPUеҸҳйҮҸдёӯзҡ„pool_workqueue гҖӮеҜ№дәҺunbound workqueueпјҢеҸ–еҮәеҪ“еүҚCPU жүҖеңЁnodeеҜ№еә”зҡ„pool_workqueueгҖӮ

еңЁkworkеҶ…ж ёиҝӣзЁӢеҶ…йғЁпјҢжңҖз»ҲдјҡеңЁеҮҪж•°process_one_workдёӯеӨ„зҗҶworkпјҡ

// kernel/workqueue.c
static void process_one_work(struct worker *worker, struct work_struct *work)
{
	// ...
	trace_workqueue_execute_start(work);	// и®°еҪ•ејҖе§ӢеӨ„зҗҶworkзҡ„trace event
	worker->current_func(work);						// и°ғз”Ёworkзҡ„еӨ„зҗҶеҮҪж•°
	trace_workqueue_execute_end(work);		// и®°еҪ•з»“жқҹеӨ„зҗҶworkзҡ„trace event
}

дҪҝз”ЁsystemtapиҝҪиёӘиҪҜдёӯж–ӯ #

systemtapдёҠжҸҗдҫӣдәҶеҸҜд»Ҙи·ҹиёӘиҪҜдёӯж–ӯд»ҘеҸҠе·ҘдҪңйҳҹеҲ—зҡ„probeдәӢд»¶пјҢеҸҜд»ҘеңЁsystemtapзҡ„tapset\linux\irq.stpдёӯзңӢеҲ°зӣёе…ізҡ„е®ҡд№үпјҢжҜ”иҫғз®ҖеҚ•иҝҷйҮҢе°ұдёҚеҒҡи§ЈйҮҠпјҢиҝҷйҮҢд»…еҲ—дёҫзҪ‘дёҠзңӢеҲ°зҡ„дёҖдёӘдҪҝз”ЁиҝҷдәӣtapsetжқҘиҝҪиёӘиҪҜдёӯж–ӯдәӢд»¶зҡ„дҫӢеӯҗпјҡ

global hard, soft, wq
 
// и®°еҪ•зЎ¬дёӯж–ӯ
probe irq_handler.entry {
  hard[irq, dev_name]++;
}
 
// жҜҸйҡ”дёҖз§’жү“еҚ°дёӯж–ӯгҖҒиҪҜдёӯж–ӯгҖҒе·ҘдҪңйҳҹеҲ—зҡ„з»ҹи®ЎдҝЎжҒҜ
probe timer.s(1) {
  println("==irq number:dev_name")
  foreach( [irq, dev_name] in hard- limit 5) {
    printf("%d,%s->%d\n", irq, kernel_string(dev_name), hard[irq, dev_name]);       
  }
 
  println("==softirq cpu:h:vec:action")
  foreach(  in soft- limit 5) {
    printf("%d:%x:%x:%s->%d\n", c, h, vec, symdata(action), soft);       
  }
 
  println("==workqueue wq_thread:work_func")
  foreach( [wq_thread,work_func] in wq- limit 5) {
    printf("%x:%x->%d\n", wq_thread, work_func, wq[wq_thread, work_func]);  
  }
 
  println("\n")
  delete hard
  delete soft
  delete wq
}
 
// з»ҹи®ЎиҪҜдёӯж–ӯ
probe softirq.entry {
  soft[cpu(), h,vec,action]++;
}
 
// з»ҹи®Ўе·ҘдҪңйҳҹеҲ—жү§иЎҢдҝЎжҒҜ
probe workqueue.execute {
  wq[wq_thread, work_func]++
}
